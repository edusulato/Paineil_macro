import matplotlib.pyplot as plt
import pandas as pd
from pathlib import Path

# 1. Carregar dados
arquivo = Path(r"C:\Users\eduar\Downloads\Tabela Inflação.xlsx")
df = pd.read_excel(
    arquivo,
    sheet_name="INFLAÇÃO",
    engine="openpyxl",
)
meses = {
    'janeiro': 1, 'fevereiro': 2, 'março': 3, 'abril': 4, 'maio': 5, 'junho': 6,
    'julho': 7, 'agosto': 8, 'setembro': 9, 'outubro': 10, 'novembro': 11, 'dezembro': 12
}
df['Mes_num'] = df['Mês'].str.lower().map(meses)
df['Data'] = pd.to_datetime(df['Ano'].astype(str) + '-' + df['Mes_num'].astype(str).str.zfill(2), format='%Y-%m')
df = df.sort_values('Data').reset_index(drop=True)

# 2. Acumulado anual da inflação
acumulado_anual = df.groupby('Ano')['Inflação'].sum().reset_index()

# 3. Encontrar ponto de separação (inflação anual <= 4%)
threshold = 4
idx_queda = acumulado_anual.index[(acumulado_anual['Ano'] >= 1994) & (acumulado_anual['Inflação'] <= threshold)][0]
anual_antes = acumulado_anual.iloc[:idx_queda]
anual_depois = acumulado_anual.iloc[idx_queda:]

# 4. Eventos históricos (só um evento por ano para evitar poluição visual)
eventos = [
    {'ano': 1986, 'label': 'Plano Cruzado'},
    {'ano': 1987, 'label': 'Plano Bresser'},
    {'ano': 1989, 'label': 'Plano Verão'},
    {'ano': 1990, 'label': 'Plano Collor I'},
    {'ano': 1991, 'label': 'Plano Collor II'},
    {'ano': 1994, 'label': 'Plano Real'},
    {'ano': 1999, 'label': 'Crise Cambial'},
    {'ano': 2001, 'label': 'Apagão'},
    {'ano': 2002, 'label': 'Eleições Lula'},
    {'ano': 2008, 'label': 'Crise Global'},
    {'ano': 2013, 'label': 'Protestos'},
    {'ano': 2015, 'label': 'Recessão'},
    {'ano': 2016, 'label': 'Impeachment Dilma'},
    {'ano': 2020, 'label': 'Covid-19'},
    {'ano': 2021, 'label': 'Crise Hídrica'},
    {'ano': 2022, 'label': 'Guerra Ucrânia'},
    {'ano': 2023, 'label': 'Mudança Política'},
]

# 5. Função para anotar eventos alternando acima/abaixo, deslocamento menor
def anotar_eventos(ax, df, eventos):
    for i, ev in enumerate(eventos):
        y = df.loc[df['Ano'] == ev['ano'], 'Inflação']
        if not y.empty:
            desloc = 5 if i%2==0 else -5
            va = 'bottom' if i%2==0 else 'top'
            ax.annotate(
                ev['label'],
                xy=(ev['ano'], y.iloc[0]),
                xytext=(ev['ano'], y.iloc[0]+desloc),
                fontsize=11,
                ha='center',
                va=va,
                arrowprops=dict(facecolor='black', arrowstyle='->', lw=1.2, alpha=0.7),
                bbox=dict(boxstyle="round,pad=0.3", fc="yellow", alpha=0.20)
            )

# 6. Gráfico 1: Antes da queda estrutural
fig, ax = plt.subplots(figsize=(16,7))
ax.plot(anual_antes['Ano'], anual_antes['Inflação'], color='darkred', linewidth=2, marker='o')
anotar_eventos(ax, anual_antes, eventos)
ax.set_title('Inflação Acumulada Anual no Brasil — Período de Hiperinflação')
ax.set_xlabel('Ano')
ax.set_ylabel('Inflação Anual (%)')
ax.grid(True, linestyle='--', alpha=0.5)
ax.set_xticks(range(anual_antes['Ano'].min(), anual_antes['Ano'].max()+1, 2))
plt.tight_layout()
plt.show()

# 7. Gráfico 2: Após a queda estrutural
fig, ax = plt.subplots(figsize=(16,7))
ax.plot(anual_depois['Ano'], anual_depois['Inflação'], color='navy', linewidth=2, marker='o')
anotar_eventos(ax, anual_depois, eventos)
ax.set_title('Inflação Acumulada Anual no Brasil — Pós-Estabilização')
ax.set_xlabel('Ano')
ax.set_ylabel('Inflação Anual (%)')
ax.grid(True, linestyle='--', alpha=0.5)
ax.set_xticks(range(anual_depois['Ano'].min(), anual_depois['Ano'].max()+1, 2))
plt.tight_layout()
plt.show()

import matplotlib.pyplot as plt
import pandas as pd
from pathlib import Path

# Eventos por ano (ordenados)
eventos_anuais = {
    1986: {'label': 'Plano Cruzado', 'desc': 'Congelamento de preços e nova moeda, seguido de perda de controle.'},
    1987: {'label': 'Plano Bresser', 'desc': 'Novo congelamento, pouco efeito duradouro sobre preços.'},
    1989: {'label': 'Plano Verão', 'desc': 'Outro plano heterodoxo; nova moeda, mas hiperinflação persiste.'},
    1990: {'label': 'Plano Collor I', 'desc': 'Confisco de poupança, choque recessivo, inflação ainda elevada.'},
    1991: {'label': 'Plano Collor II', 'desc': 'Nova tentativa, sem sucesso, de estabilizar preços.'},
    1994: {'label': 'Plano Real', 'desc': 'Estabilização duradoura: URV, nova moeda, queda histórica da inflação.'},
    1999: {'label': 'Crise Cambial', 'desc': 'Fim do câmbio fixo, real desvaloriza e inflação repica.'},
    2001: {'label': 'Apagão', 'desc': 'Racionamento de energia, impacto moderado em preços.'},
    2002: {'label': 'Eleições Lula', 'desc': 'Risco-Brasil alto, real se desvaloriza e preços aceleram.'},
    2008: {'label': 'Crise Global', 'desc': 'Crise internacional, impacto temporário e repique nos preços.'},
    2013: {'label': 'Protestos', 'desc': 'Inflação de alimentos e transportes, contexto social instável.'},
    2015: {'label': 'Recessão', 'desc': 'Choque de preços administrados, recessão profunda.'},
    2016: {'label': 'Impeachment Dilma', 'desc': 'Cenário recessivo e pressão política, preços perdem força.'},
    2020: {'label': 'Covid-19', 'desc': 'Inflação de alimentos, câmbio, choque global da pandemia.'},
    2021: {'label': 'Crise Hídrica', 'desc': 'Energia mais cara e impacto em alimentos.'},
    2022: {'label': 'Guerra Ucrânia', 'desc': 'Choque de petróleo e alimentos global.'},
    2023: {'label': 'Mudança Política', 'desc': 'Expectativa de mudança de governo, cenário moderado.'},
    2024: {'label': 'Ano eleitoral, pós-covid', 'desc': 'Inflação sob controle, porém pressão de alimentos e serviços.'},
}
anos_evento = sorted(eventos_anuais.keys())

def proximo_evento_relevante(ano):
    for future_year in anos_evento:
        if future_year > ano:
            ev = eventos_anuais[future_year]
            return f"Próximo evento: {future_year}, {ev['label']}: {ev['desc']}"
    return "Sem evento marcante próximo."

def linha_evento(ano):
    if ano in eventos_anuais:
        ev = eventos_anuais[ano]
        return f"{ev['label']}: {ev['desc']}"
    else:
        return proximo_evento_relevante(ano)

# Dados
arquivo = Path(r"C:\Users\eduar\Downloads\Tabela Inflação.xlsx")
df = pd.read_excel(arquivo, sheet_name="INFLAÇÃO", engine="openpyxl")
df['Mes_num'] = df['Mês'].str.lower().map({
    'janeiro': 1, 'fevereiro': 2, 'março': 3, 'abril': 4, 'maio': 5, 'junho': 6,
    'julho': 7, 'agosto': 8, 'setembro': 9, 'outubro': 10, 'novembro': 11, 'dezembro': 12
})
df['Data'] = pd.to_datetime(df['Ano'].astype(str) + '-' + df['Mes_num'].astype(str).str.zfill(2), format='%Y-%m')
df = df.sort_values('Data').reset_index(drop=True)

# Acumulado anual - SOMENTE ATÉ 2024!
acumulado_anual = df.groupby('Ano')['Inflação'].sum().reset_index()
acumulado_anual = acumulado_anual[acumulado_anual['Ano'] <= 2024]
top5_altas = acumulado_anual.nlargest(5, 'Inflação')
top5_quedas = acumulado_anual.nsmallest(5, 'Inflação')

# Inflação atual
df_2024 = df[df['Ano'] == 2024]
inflacao_acum_2024 = df_2024['Inflação'].sum()
ultimo_mes_2024 = df_2024['Mês'].iloc[-1].capitalize() if not df_2024.empty else 'N/A'

macro = (
    "Em 2024, a inflação brasileira segue sob controle, fechando o acumulado anual em patamar historicamente baixo, "
    f"mesmo após choques recentes em alimentos e combustíveis. O Banco Central mantém a Selic elevada, ancorando expectativas. "
    "Alimentos subiram pontualmente após eventos climáticos (seca no RS), mas núcleos de inflação continuam baixos. "
    "O desemprego segue em queda moderada e a política fiscal estável reduz riscos de repique inflacionário. "
    "No cenário externo, commodities e juros americanos ainda são fonte de volatilidade, mas a desinflação global ajuda o Brasil."
)

f_titulo = 17

block_offset = 0.23  # distância de um bloco para outro

for i, row in enumerate(top5_altas.itertuples()):
    explic = linha_evento(int(row.Ano))
    plt.text(0.04, top_altas_y-0.055-(i*dy), f"{int(row.Ano)}: {row.Inflação:.1f}%  — {explic}", ha="left", va="center", fontsize=f_lista, color="#B20B0B")

top_quedas_y = top_altas_y - (block_offset)

plt.text(0.03, top_quedas_y, "Top 5 Menores Altas Anuais", ha="left", va="center", fontsize=f_bloco, fontweight="bold", color="#145A99")
for i, row in enumerate(top5_quedas.itertuples()):
    explic = linha_evento(int(row.Ano))
    plt.text(0.04, top_quedas_y-0.055-(i*dy), f"{int(row.Ano)}: {row.Inflação:.1f}%  — {explic}", ha="left", va="center", fontsize=f_lista, color="#0A4383")

# Acumulado e macro
acum_y = top_quedas_y - (block_offset*1.05)
plt.gca().add_patch(plt.Rectangle((0, acum_y-0.03), 1, 0.04, color="#FFF7E6", zorder=1))
