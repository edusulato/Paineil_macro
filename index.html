import matplotlib.pyplot as plt
import pandas as pd
from pathlib import Path

# 1. Carregar dados
arquivo = Path(r"C:\Users\eduar\Downloads\Tabela Inflao.xlsx")
df = pd.read_excel(
    arquivo,
    sheet_name="INFLAO",
    engine="openpyxl",
)
meses = {
    'janeiro': 1, 'fevereiro': 2, 'maro': 3, 'abril': 4, 'maio': 5, 'junho': 6,
    'julho': 7, 'agosto': 8, 'setembro': 9, 'outubro': 10, 'novembro': 11, 'dezembro': 12
}
df['Mes_num'] = df['Ms'].str.lower().map(meses)
df['Data'] = pd.to_datetime(df['Ano'].astype(str) + '-' + df['Mes_num'].astype(str).str.zfill(2), format='%Y-%m')
df = df.sort_values('Data').reset_index(drop=True)

# 2. Acumulado anual da inflao
acumulado_anual = df.groupby('Ano')['Inflao'].sum().reset_index()

# 3. Encontrar ponto de separao (inflao anual <= 4%)
threshold = 4
idx_queda = acumulado_anual.index[(acumulado_anual['Ano'] >= 1994) & (acumulado_anual['Inflao'] <= threshold)][0]
anual_antes = acumulado_anual.iloc[:idx_queda]
anual_depois = acumulado_anual.iloc[idx_queda:]

# 4. Eventos histricos (s um evento por ano para evitar poluio visual)
eventos = [
    {'ano': 1986, 'label': 'Plano Cruzado'},
    {'ano': 1987, 'label': 'Plano Bresser'},
    {'ano': 1989, 'label': 'Plano Vero'},
    {'ano': 1990, 'label': 'Plano Collor I'},
    {'ano': 1991, 'label': 'Plano Collor II'},
    {'ano': 1994, 'label': 'Plano Real'},
    {'ano': 1999, 'label': 'Crise Cambial'},
    {'ano': 2001, 'label': 'Apago'},
    {'ano': 2002, 'label': 'Eleies Lula'},
    {'ano': 2008, 'label': 'Crise Global'},
    {'ano': 2013, 'label': 'Protestos'},
    {'ano': 2015, 'label': 'Recesso'},
    {'ano': 2016, 'label': 'Impeachment Dilma'},
    {'ano': 2020, 'label': 'Covid-19'},
    {'ano': 2021, 'label': 'Crise Hdrica'},
    {'ano': 2022, 'label': 'Guerra Ucrnia'},
    {'ano': 2023, 'label': 'Mudana Poltica'},
]

# 5. Funo para anotar eventos alternando acima/abaixo, deslocamento menor
def anotar_eventos(ax, df, eventos):
    for i, ev in enumerate(eventos):
        y = df.loc[df['Ano'] == ev['ano'], 'Inflao']
        if not y.empty:
            desloc = 5 if i%2==0 else -5
            va = 'bottom' if i%2==0 else 'top'
            ax.annotate(
                ev['label'],
                xy=(ev['ano'], y.iloc[0]),
                xytext=(ev['ano'], y.iloc[0]+desloc),
                fontsize=11,
                ha='center',
                va=va,
                arrowprops=dict(facecolor='black', arrowstyle='->', lw=1.2, alpha=0.7),
                bbox=dict(boxstyle="round,pad=0.3", fc="yellow", alpha=0.20)
            )

# 6. Grfico 1: Antes da queda estrutural
fig, ax = plt.subplots(figsize=(16,7))
ax.plot(anual_antes['Ano'], anual_antes['Inflao'], color='darkred', linewidth=2, marker='o')
anotar_eventos(ax, anual_antes, eventos)
ax.set_title('Inflao Acumulada Anual no Brasil  Perodo de Hiperinflao')
ax.set_xlabel('Ano')
ax.set_ylabel('Inflao Anual (%)')
ax.grid(True, linestyle='--', alpha=0.5)
ax.set_xticks(range(anual_antes['Ano'].min(), anual_antes['Ano'].max()+1, 2))
plt.tight_layout()
plt.show()

# 7. Grfico 2: Aps a queda estrutural
fig, ax = plt.subplots(figsize=(16,7))
ax.plot(anual_depois['Ano'], anual_depois['Inflao'], color='navy', linewidth=2, marker='o')
anotar_eventos(ax, anual_depois, eventos)
ax.set_title('Inflao Acumulada Anual no Brasil  Ps-Estabilizao')
ax.set_xlabel('Ano')
ax.set_ylabel('Inflao Anual (%)')
ax.grid(True, linestyle='--', alpha=0.5)
ax.set_xticks(range(anual_depois['Ano'].min(), anual_depois['Ano'].max()+1, 2))
plt.tight_layout()
plt.show()

import matplotlib.pyplot as plt
import pandas as pd
from pathlib import Path

# Eventos por ano (ordenados)
eventos_anuais = {
    1986: {'label': 'Plano Cruzado', 'desc': 'Congelamento de preos e nova moeda, seguido de perda de controle.'},
    1987: {'label': 'Plano Bresser', 'desc': 'Novo congelamento, pouco efeito duradouro sobre preos.'},
    1989: {'label': 'Plano Vero', 'desc': 'Outro plano heterodoxo; nova moeda, mas hiperinflao persiste.'},
    1990: {'label': 'Plano Collor I', 'desc': 'Confisco de poupana, choque recessivo, inflao ainda elevada.'},
    1991: {'label': 'Plano Collor II', 'desc': 'Nova tentativa, sem sucesso, de estabilizar preos.'},
    1994: {'label': 'Plano Real', 'desc': 'Estabilizao duradoura: URV, nova moeda, queda histrica da inflao.'},
    1999: {'label': 'Crise Cambial', 'desc': 'Fim do cmbio fixo, real desvaloriza e inflao repica.'},
    2001: {'label': 'Apago', 'desc': 'Racionamento de energia, impacto moderado em preos.'},
    2002: {'label': 'Eleies Lula', 'desc': 'Risco-Brasil alto, real se desvaloriza e preos aceleram.'},
    2008: {'label': 'Crise Global', 'desc': 'Crise internacional, impacto temporrio e repique nos preos.'},
    2013: {'label': 'Protestos', 'desc': 'Inflao de alimentos e transportes, contexto social instvel.'},
    2015: {'label': 'Recesso', 'desc': 'Choque de preos administrados, recesso profunda.'},
    2016: {'label': 'Impeachment Dilma', 'desc': 'Cenrio recessivo e presso poltica, preos perdem fora.'},
    2020: {'label': 'Covid-19', 'desc': 'Inflao de alimentos, cmbio, choque global da pandemia.'},
    2021: {'label': 'Crise Hdrica', 'desc': 'Energia mais cara e impacto em alimentos.'},
    2022: {'label': 'Guerra Ucrnia', 'desc': 'Choque de petrleo e alimentos global.'},
    2023: {'label': 'Mudana Poltica', 'desc': 'Expectativa de mudana de governo, cenrio moderado.'},
    2024: {'label': 'Ano eleitoral, ps-covid', 'desc': 'Inflao sob controle, porm presso de alimentos e servios.'},
}
anos_evento = sorted(eventos_anuais.keys())

def proximo_evento_relevante(ano):
    for future_year in anos_evento:
        if future_year > ano:
            ev = eventos_anuais[future_year]
            return f"Prximo evento: {future_year}, {ev['label']}: {ev['desc']}"
    return "Sem evento marcante prximo."

def linha_evento(ano):
    if ano in eventos_anuais:
        ev = eventos_anuais[ano]
        return f"{ev['label']}: {ev['desc']}"
    else:
        return proximo_evento_relevante(ano)

# Dados
arquivo = Path(r"C:\Users\eduar\Downloads\Tabela Inflao.xlsx")
df = pd.read_excel(arquivo, sheet_name="INFLAO", engine="openpyxl")
df['Mes_num'] = df['Ms'].str.lower().map({
    'janeiro': 1, 'fevereiro': 2, 'maro': 3, 'abril': 4, 'maio': 5, 'junho': 6,
    'julho': 7, 'agosto': 8, 'setembro': 9, 'outubro': 10, 'novembro': 11, 'dezembro': 12
})
df['Data'] = pd.to_datetime(df['Ano'].astype(str) + '-' + df['Mes_num'].astype(str).str.zfill(2), format='%Y-%m')
df = df.sort_values('Data').reset_index(drop=True)

# Acumulado anual - SOMENTE AT 2024!
acumulado_anual = df.groupby('Ano')['Inflao'].sum().reset_index()
acumulado_anual = acumulado_anual[acumulado_anual['Ano'] <= 2024]
top5_altas = acumulado_anual.nlargest(5, 'Inflao')
top5_quedas = acumulado_anual.nsmallest(5, 'Inflao')

# Inflao atual
df_2024 = df[df['Ano'] == 2024]
inflacao_acum_2024 = df_2024['Inflao'].sum()
ultimo_mes_2024 = df_2024['Ms'].iloc[-1].capitalize() if not df_2024.empty else 'N/A'

macro = (
    "Em 2024, a inflao brasileira segue sob controle, fechando o acumulado anual em patamar historicamente baixo, "
    f"mesmo aps choques recentes em alimentos e combustveis. O Banco Central mantm a Selic elevada, ancorando expectativas. "
    "Alimentos subiram pontualmente aps eventos climticos (seca no RS), mas ncleos de inflao continuam baixos. "
    "O desemprego segue em queda moderada e a poltica fiscal estvel reduz riscos de repique inflacionrio. "
    "No cenrio externo, commodities e juros americanos ainda so fonte de volatilidade, mas a desinflao global ajuda o Brasil."
)

# --- Painel COMPACTO ---
fig, ax = plt.subplots(figsize=(8, 15))  # imagem menor!

plt.xlim(0, 1)
plt.ylim(0, 1)

# Parmetros proporcionais (menores fontes, mesmos espaos)
f_titulo = 17
f_bloco = 12
f_lista = 10
dy = 0.037  # espao vertical por linha
block_offset = 0.23  # distncia de um bloco para outro

# Ttulo
plt.text(0.5, 0.97, "Painel Matinal de Inflao (Brasil)", ha="center", va="center", fontsize=f_titulo, fontweight="bold", color="#222")

# Maiores altas
top_altas_y = 0.92
plt.gca().add_patch(plt.Rectangle((0, top_altas_y-0.04), 1, 0.047, color="#FFDD99", zorder=1))
plt.text(0.03, top_altas_y, "Top 5 Maiores Altas Anuais", ha="left", va="center", fontsize=f_bloco, fontweight="bold", color="#B18800")
for i, row in enumerate(top5_altas.itertuples()):
    explic = linha_evento(int(row.Ano))
    plt.text(0.04, top_altas_y-0.055-(i*dy), f"{int(row.Ano)}: {row.Inflao:.1f}%   {explic}", ha="left", va="center", fontsize=f_lista, color="#B20B0B")

# Maiores quedas
top_quedas_y = top_altas_y - (block_offset)
plt.gca().add_patch(plt.Rectangle((0, top_quedas_y-0.04), 1, 0.047, color="#BBDEFB", zorder=1))
plt.text(0.03, top_quedas_y, "Top 5 Menores Altas Anuais", ha="left", va="center", fontsize=f_bloco, fontweight="bold", color="#145A99")
for i, row in enumerate(top5_quedas.itertuples()):
    explic = linha_evento(int(row.Ano))
    plt.text(0.04, top_quedas_y-0.055-(i*dy), f"{int(row.Ano)}: {row.Inflao:.1f}%   {explic}", ha="left", va="center", fontsize=f_lista, color="#0A4383")

# Acumulado e macro
acum_y = top_quedas_y - (block_offset*1.05)
plt.gca().add_patch(plt.Rectangle((0, acum_y-0.03), 1, 0.04, color="#FFF7E6", zorder=1))
plt.text(0.03, acum_y, f"Inflao acumulada em 2024 (at {ultimo_mes_2024}):", ha="left", va="center", fontsize=f_bloco, color="#444")

# Resumo macro
macro_y = acum_y - 0.07
plt.text(0.03, macro_y, "Resumo Macro Atual:", ha="left", va="center", fontsize=f_bloco, color="#B18800", fontweight="bold")
plt.text(0.03, macro_y-0.025, macro, ha="left", va="top", fontsize=f_lista, color="#333", wrap=True)
